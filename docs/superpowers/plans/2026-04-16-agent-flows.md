# Agent Flows + Launcher + Enforcement Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** User clicks "+ Flow" in the dashboard, types a feature description, and watches a capataz orchestrate a team of role-specific agents (architect, backend, frontend, tester, …) with hierarchy visualized, `runsAfter` enforced, and token-safe timeouts.

**Architecture:** Five new core modules — `flowsConfig` (read/merge config), `flowsConfig.builtin` (hardcoded defaults), `capatazPromptBuilder` (pure prompt builder), `runsAfterGate` (PreToolUse interceptor with pending-queue drain on SubagentStop), `timeoutWatchdog` (per-agent idle+total kill) — plus an extended `Orchestrator.handleLaunchFlow()` and new webview components (`LaunchFlowModal`, team-line renderer, cian waiting bubble, role labels). Teammate detection reuses existing `claudeTeamProvider` + `HookEventHandler` paths.

**Tech Stack:** TypeScript (core + webview), vitest (unit + integration tests), React (modal + settings toggle), pngjs (cian bubble sprite pre-baked via existing scripts/generate-cave-theme.ts pattern).

**Spec:** `docs/superpowers/specs/2026-04-16-agent-flows-dashboard-design.md`.

**Depends on:** `docs/superpowers/plans/2026-04-16-passive-dashboard-hooks.md` — the hook server + `HookEventHandler` routing must be wired to the Orchestrator. `runsAfterGate` plugs into the same hook pipeline.

```typescript
import * as fs from 'node:fs';
import * as path from 'node:path';
import { BUILTIN_FLOWS_CONFIG } from './flowsConfig.builtin.js';

export interface LoadContext {
  homeDir: string;
  projectDir: string;
}

const GLOBAL_REL = path.join('.pixel-agents', 'flows.json');
const PROJECT_REL = path.join('.pixel-agents', 'flows.json');

/** Read a single flows.json file. Returns null on missing/malformed. */
function readLayer(filePath: string): Partial<FlowsConfig> | null {
  try {
    const raw = fs.readFileSync(filePath, 'utf-8');
    return JSON.parse(raw) as Partial<FlowsConfig>;
  } catch {
    return null;
  }
}

/**
 * Merge `b` on top of `a`. Rules:
 * - `roles`: by key, b's record wins entirely (NO deep merge of fields).
 * - `flows`: by `id`, b's record wins entirely. New ids from b appended.
 * - `defaults`: shallow merge (b overrides a at top level).
 */
function mergeConfigs(a: FlowsConfig, b: Partial<FlowsConfig>): FlowsConfig {
  const roles = { ...a.roles, ...(b.roles ?? {}) };
  const flowsById = new Map(a.flows.map((f) => [f.id, f]));
  for (const f of b.flows ?? []) flowsById.set(f.id, f);
  const flows = [...flowsById.values()];
  const defaults = { ...a.defaults, ...(b.defaults ?? {}) };
  return { roles, flows, defaults };
}

export function loadFlowsConfig(ctx: LoadContext): FlowsConfig {
  const globalPath = path.join(ctx.homeDir, GLOBAL_REL);
  const projectPath = path.join(ctx.projectDir, PROJECT_REL);
  let cfg = BUILTIN_FLOWS_CONFIG;
  const g = readLayer(globalPath);
  if (g) cfg = mergeConfigs(cfg, g);
  const p = readLayer(projectPath);
  if (p) cfg = mergeConfigs(cfg, p);
  return cfg;
}
```

```typescript
/**
 * Detect cycles in a directed graph where each node has 0+ outgoing edges.
 * Uses DFS with a visiting/visited set. Returns the first cycle found (node ids in order).
 */
function findCycle(
  nodes: string[],
  outgoing: (id: string) => string[],
): string[] | null {
  const WHITE = 0, GRAY = 1, BLACK = 2;
  const color = new Map<string, number>(nodes.map((n) => [n, WHITE]));
  const stack: string[] = [];

  function dfs(n: string): string[] | null {
    color.set(n, GRAY);
    stack.push(n);
    for (const next of outgoing(n)) {
      if (!color.has(next)) continue;  // dangling edges handled by missingRole check
      const c = color.get(next)!;
      if (c === GRAY) {
        // cycle from `next` back to current path
        const cycleStart = stack.indexOf(next);
        return stack.slice(cycleStart).concat(next);
      }
      if (c === WHITE) {
        const found = dfs(next);
        if (found) return found;
      }
    }
    color.set(n, BLACK);
    stack.pop();
    return null;
  }

  for (const n of nodes) {
    if (color.get(n) === WHITE) {
      const found = dfs(n);
      if (found) return found;
    }
  }
  return null;
}

export function validateConfig(config: FlowsConfig): ValidationError[] {
  const errors: ValidationError[] = [];
  const roleIds = Object.keys(config.roles);

  // Missing roles referenced by flows
  for (const f of config.flows) {
    for (const r of f.roles) {
      if (!(r in config.roles)) {
        errors.push({
          kind: 'missingRole',
          message: `Flow "${f.id}" references missing role "${r}"`,
          path: `flows[${f.id}].roles`,
        });
      }
    }
  }

  // Duplicate flow ids
  const seenFlowIds = new Set<string>();
  for (const f of config.flows) {
    if (seenFlowIds.has(f.id)) {
      errors.push({
        kind: 'duplicateId',
        message: `Duplicate flow id "${f.id}"`,
        path: `flows[${f.id}]`,
      });
    }
    seenFlowIds.add(f.id);
  }

  // runsAfter cycle
  const runsAfterCycle = findCycle(roleIds, (id) => config.roles[id]?.runsAfter ?? []);
  if (runsAfterCycle) {
    errors.push({
      kind: 'cycle',
      message: `runsAfter cycle: ${runsAfterCycle.join(' → ')}`,
      path: 'roles.*.runsAfter',
    });
  }

  // reportsTo cycle
  const reportsToCycle = findCycle(
    roleIds,
    (id) => (config.roles[id]?.reportsTo ? [config.roles[id].reportsTo!] : []),
  );
  if (reportsToCycle) {
    errors.push({
      kind: 'cycle',
      message: `reportsTo cycle: ${reportsToCycle.join(' → ')}`,
      path: 'roles.*.reportsTo',
    });
  }

  // blocksOn cycle
  const blocksOnCycle = findCycle(roleIds, (id) => config.roles[id]?.blocksOn ?? []);
  if (blocksOnCycle) {
    errors.push({
      kind: 'cycle',
      message: `blocksOn cycle: ${blocksOnCycle.join(' → ')}`,
      path: 'roles.*.blocksOn',
    });
  }

  return errors;
}
```

```typescript
     entre ellos.
  6. Si recibís un mensaje que empieza con '[sistema]' — es automático del runtime. Respondé

function formatRoleLine(id: string, role: Role): string {
  const relations: string[] = [];
  if (role.supervises) relations.push(`supervises: [${role.supervises.join(', ')}]`);
  if (role.reportsTo) relations.push(`reports_to: ${role.reportsTo}`);
  if (role.runsAfter) relations.push(`runs_after: [${role.runsAfter.join(', ')}]`);
  if (role.blocksOn) relations.push(`blocks_on: [${role.blocksOn.join(', ')}]`);
  if (role.collaboratesWith) relations.push(`collaborates_with: [${role.collaboratesWith.join(', ')}]`);
  const icon = role.icon ? `${role.icon}  ` : '';
  const relStr = relations.length > 0 ? `  ${relations.join(', ')}` : '';
  return `  ${id.padEnd(12)}${icon}${role.label}${relStr}\n    → ${role.prompt}`;
}

function buildRoleLibraryBlock(config: FlowsConfig): string {
  const lines = ['ROLES DISPONIBLES:'];
  for (const [id, role] of Object.entries(config.roles)) {
    lines.push(formatRoleLine(id, role));
  }
  return lines.join('\n');
}

function interpolate(template: string, vars: Record<string, string>): string {
  return template.replace(/\{\{(\w+)\}\}/g, (match, key) => {
    return key in vars ? vars[key] : match;
  });
}

export function buildCapatazPrompt(
  flow: Flow,
  config: FlowsConfig,
  userInput: string,
  context: BuildContext,
): BuildResult {
  const vars = {
    userInput,
    project: context.project,
    branch: context.branch ?? '',
  };
  const systemPrompt = [
    HEADER,
    '',
    buildRoleLibraryBlock(config),
    '',
    EXECUTION_RULES_TEMPLATE,
  ].join('\n');
  const userPrompt = interpolate(flow.promptTemplate, vars);
  return { systemPrompt, userPrompt };
}
```

```typescript
// core/src/runsAfterGate.ts
import type { FlowsConfig } from './flowsConfig.js';

export interface PreToolUseEvent {
  toolName: string;
  /** The role id being spawned (from Agent tool `description` param). */
  role: string;
  /** Session id of the capataz making the call. */
  capatazSessionId: string;
}

export interface SubagentStopEvent {
  capatazSessionId: string;
  role: string;
}

export interface GateDecision {
  deny: boolean;
  reason?: string;
}

export interface GateContext {
  config: FlowsConfig;
  /** Roles currently active (spawned but not yet stopped) under this capataz. */
  activeTeammateRoles: Set<string>;
  /** Roles that have finished for this capataz (used for drain). */
  finishedTeammateRoles: Set<string>;
  /** Callback fired when a role has been denied 3+ times. */
  onFlowBlocked?: (info: { capatazSessionId: string; blockedRole: string; attempts: number }) => void;
  /** Callback fired on drain — capataz should re-attempt the spawn. */
  sendSystemMessageToCapataz?: (capatazSessionId: string, message: string) => void;
}

interface PendingEntry {
  role: string;
  attempts: number;
}

const DEADLOCK_THRESHOLD = 3;

export class RunsAfterGate {
  /** key: capatazSessionId → pending spawn entries. */
  private pending = new Map<string, PendingEntry[]>();

  /**
   * Inspect a PreToolUse call. Returns GateDecision when the call is relevant
   * (an `Agent` for a role with `runsAfter` deps); null otherwise.
   */
  onPreToolUse(event: PreToolUseEvent, ctx: GateContext): GateDecision | null {
    if (event.toolName !== 'Agent') return null;
    const role = ctx.config.roles[event.role];
    if (!role?.runsAfter || role.runsAfter.length === 0) return null;

    const unfinishedDeps = role.runsAfter.filter(
      (dep) => !ctx.finishedTeammateRoles.has(dep),
    );
    if (unfinishedDeps.length === 0) {
      // Deps satisfied — pop from pending if present.
      this.removePending(event.capatazSessionId, event.role);
      return { deny: false };
    }

    // Deny + record.
    const entry = this.addOrIncrement(event.capatazSessionId, event.role);
    if (entry.attempts >= DEADLOCK_THRESHOLD) {
      ctx.onFlowBlocked?.({
        capatazSessionId: event.capatazSessionId,
        blockedRole: event.role,
        attempts: entry.attempts,
      });
    }
    return {
      deny: true,
      reason: `Role "${event.role}" requires [${role.runsAfter.join(', ')}] to finish first. Still active: [${unfinishedDeps.join(', ')}]. Wait for '[sistema] ${role.runsAfter.join(' + ')} completaron' signal before retrying.`,
    };
  }

  /**
   * When a teammate stops (SubagentStop hook), drain any pending spawns whose
   * runsAfter deps are now all finished.
   */
  onSubagentStop(event: SubagentStopEvent, ctx: GateContext): void {
    const pending = this.pending.get(event.capatazSessionId) ?? [];
    const stillPending: PendingEntry[] = [];
    for (const entry of pending) {
      const role = ctx.config.roles[entry.role];
      if (!role?.runsAfter) continue;
      const unfinishedDeps = role.runsAfter.filter(
        (dep) => !ctx.finishedTeammateRoles.has(dep),
      );
      if (unfinishedDeps.length === 0) {
        // Drain: notify capataz to retry.
        ctx.sendSystemMessageToCapataz?.(
          event.capatazSessionId,
          `[sistema] ${role.runsAfter.join(' + ')} completaron. Proceder con ${entry.role}.`,
        );
        // Don't add to stillPending.
      } else {
        stillPending.push(entry);
      }
    }
    if (stillPending.length === 0) {
      this.pending.delete(event.capatazSessionId);
    } else {
      this.pending.set(event.capatazSessionId, stillPending);
    }
  }

  /** Force-allow a specific spawn (user clicked "Forzar spawn"). */
  forceSpawn(capatazSessionId: string, role: string): void {
    this.removePending(capatazSessionId, role);
  }

  private addOrIncrement(capatazSessionId: string, role: string): PendingEntry {
    let list = this.pending.get(capatazSessionId);
    if (!list) {
      list = [];
      this.pending.set(capatazSessionId, list);
    }
    const existing = list.find((e) => e.role === role);
    if (existing) {
      existing.attempts += 1;
      return existing;
    }
    const entry: PendingEntry = { role, attempts: 1 };
    list.push(entry);
    return entry;
  }

  private removePending(capatazSessionId: string, role: string): void {
    const list = this.pending.get(capatazSessionId);
    if (!list) return;
    const remaining = list.filter((e) => e.role !== role);
    if (remaining.length === 0) this.pending.delete(capatazSessionId);
    else this.pending.set(capatazSessionId, remaining);
  }
}
```

```typescript
// core/src/timeoutWatchdog.ts
export interface TimeoutConfig {
  idleMs: number;
  totalMs: number;
}

export interface TimeoutEvent {
  agentId: number;
  reason: 'idle' | 'total';
}

export interface WarningEvent {
  agentId: number;
  remainingMs: number;
}

export interface WatchdogOptions {
  onTimeout: (event: TimeoutEvent) => void;
  onWarning?: (event: WarningEvent) => void;
  /** Poll interval in ms. Production: 10000. Tests: override to e.g. 100. */
  checkIntervalMs?: number;
}

interface WatchedAgent {
  spawnedAt: number;
  lastActivityAt: number;
  idleMs: number;
  totalMs: number;
  warned: boolean;
}

const WARNING_LEAD_MS = 30_000;

export class TimeoutWatchdog {
  private watched = new Map<number, WatchedAgent>();
  private timer: ReturnType<typeof setInterval> | null = null;

  constructor(private readonly opts: WatchdogOptions) {
    this.timer = setInterval(() => this.tick(), opts.checkIntervalMs ?? 10_000);
  }

  start(agentId: number, config: TimeoutConfig): void {
    const now = Date.now();
    this.watched.set(agentId, {
      spawnedAt: now,
      lastActivityAt: now,
      idleMs: config.idleMs,
      totalMs: config.totalMs,
      warned: false,
    });
  }

  touch(agentId: number): void {
    const agent = this.watched.get(agentId);
    if (!agent) return;
    agent.lastActivityAt = Date.now();
    agent.warned = false;
  }

  stop(agentId: number): void {
    this.watched.delete(agentId);
  }

  extendTimeout(agentId: number, extraMs: number): void {
    const agent = this.watched.get(agentId);
    if (!agent) return;
    agent.lastActivityAt = Date.now() + extraMs - agent.idleMs;
    agent.warned = false;
  }

  dispose(): void {
    if (this.timer) clearInterval(this.timer);
    this.timer = null;
    this.watched.clear();
  }

  private tick(): void {
    const now = Date.now();
    for (const [agentId, agent] of this.watched) {
      const idleElapsed = now - agent.lastActivityAt;
      const totalElapsed = now - agent.spawnedAt;

      if (totalElapsed >= agent.totalMs) {
        this.opts.onTimeout({ agentId, reason: 'total' });
        this.watched.delete(agentId);
        continue;
      }
      if (idleElapsed >= agent.idleMs) {
        this.opts.onTimeout({ agentId, reason: 'idle' });
        this.watched.delete(agentId);
        continue;
      }
      if (
        !agent.warned &&
        agent.idleMs > WARNING_LEAD_MS &&
        idleElapsed >= agent.idleMs - WARNING_LEAD_MS
      ) {
        agent.warned = true;
        this.opts.onWarning?.({ agentId, remainingMs: agent.idleMs - idleElapsed });
      }
    }
  }
}
```

```typescript
    const bypass =
```

```typescript
  private onAgentTimeout(agentId: number, reason: 'idle' | 'total'): void {
```

```typescript
function renderTeamLines(ctx: CanvasRenderingContext2D, state: OfficeState): void {
  if (!state.showTeamLines) return;
  ctx.save();
  for (const [, ch] of state.characters) {
    if (!ch.roleRelations) continue;
    const from = { x: ch.x, y: ch.y };
    // reportsTo: solid gold line up to the supervisor.
    if (ch.roleRelations.reportsTo) {
      const sup = findCharByRole(state, ch.roleRelations.reportsTo);
      if (sup) {
        ctx.strokeStyle = '#DAA520';
        ctx.lineWidth = 1;
        ctx.beginPath();
        ctx.moveTo(from.x, from.y);
        ctx.lineTo(sup.x, sup.y);
        ctx.stroke();
      }
    }
    // collaboratesWith: dotted gold line.
    for (const peer of ch.roleRelations.collaboratesWith ?? []) {
      const p = findCharByRole(state, peer);
      if (!p) continue;
      ctx.strokeStyle = '#DAA520';
      ctx.setLineDash([2, 2]);
      ctx.beginPath();
      ctx.moveTo(from.x, from.y);
      ctx.lineTo(p.x, p.y);
      ctx.stroke();
      ctx.setLineDash([]);
    }
    // runsAfter: arrow from dep to dependent (we draw INCOMING arrow on the dependent side).
    for (const dep of ch.roleRelations.runsAfter ?? []) {
      const d = findCharByRole(state, dep);
      if (!d) continue;
      ctx.strokeStyle = '#DAA520';
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.moveTo(d.x, d.y);
      ctx.lineTo(from.x, from.y);
      ctx.stroke();
      // Small chevron arrowhead at `from`.
      drawChevron(ctx, from, angleBetween(d, from), '#DAA520');
    }
  }
  ctx.restore();
}

function findCharByRole(state: OfficeState, roleId: string) {
  for (const [, ch] of state.characters) {
    if (ch.roleId === roleId) return ch;
  }
  return null;
}

function drawChevron(
  ctx: CanvasRenderingContext2D,
  at: { x: number; y: number },
  angleRad: number,
  color: string,
): void {
  const size = 3;
  ctx.fillStyle = color;
  const dx = Math.cos(angleRad) * size;
  const dy = Math.sin(angleRad) * size;
  ctx.beginPath();
  ctx.moveTo(at.x, at.y);
  ctx.lineTo(at.x - dx + dy / 2, at.y - dy - dx / 2);
  ctx.lineTo(at.x - dx - dy / 2, at.y - dy + dx / 2);
  ctx.closePath();
  ctx.fill();
}

function angleBetween(from: { x: number; y: number }, to: { x: number; y: number }): number {
  return Math.atan2(to.y - from.y, to.x - from.x);
}
```

Call `renderTeamLines(ctx, state)` in the main render loop, after tiles and before characters.
